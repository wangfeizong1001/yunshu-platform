## 二、核心配置

### 2.1 i18n 初始化

```typescript
// apps/admin/src/locales/index.ts

import { createI18n } from 'vue-i18n'

// 语言包
import zhCN from './zh-CN'
import enUS from './en'

/**
 * 支持的语言列表
 */
export const SUPPORT_LOCALES = [
  { code: 'zh-CN', name: '简体中文', flag: '🇨🇳' },
  { code: 'en', name: 'English', flag: '🇺🇸' }
] as const

/** 默认语言 */
export const DEFAULT_LOCALE = 'zh-CN'

/** 语言存储 Key */
const LOCALE_STORAGE_KEY = 'app-locale'

/**
 * 获取当前语言（从 localStorage 读取，无则用默认）
 */
export function getLocale(): string {
  const saved = localStorage.getItem(LOCALE_STORAGE_KEY)
  if (saved && SUPPORT_LOCALES.some((l) => l.code === saved)) {
    return saved
  }
  // 尝试浏览器语言
  const browserLang = navigator.language
  if (browserLang.startsWith('zh')) return 'zh-CN'
  return 'en'
}

/**
 * 切换语言
 */
export function setLocale(locale: string) {
  localStorage.setItem(LOCALE_STORAGE_KEY, locale)
  i18n.global.locale.value = locale as any

  // 同步 Element Plus 语言（如有需要）
  // document.documentElement.lang = locale
}

/**
 * i18n 实例
 */
export const i18n = createI18n({
  legacy: false, // 使用 Composition API 模式
  locale: getLocale(),
  fallbackLocale: 'en', // 缺失键时的回退语言
  globalInjection: true, // 全局注入 $t
  messages: {
    'zh-CN': zhCN,
    en: enUS
  },
  // 缺失键处理
  missing: (locale, key) => {
    if (import.meta.env.DEV) {
      console.warn(`[i18n] 缺失翻译: ${locale}.${key}`)
    }
    return key
  }
})

export default i18n
```

### 2.2 在应用中注册

```typescript
// apps/admin/src/main.ts
```

## 三、语言包编写规范

### 3.1 Key 命名约定

```
格式: {模块}.{子模块}.{组件}.{动作/名词}

示例:
  - common.operation.confirm
  - common.operation.cancel
  - common.operation.save
  - system.user.title
  - system.user.search.placeholder
  - system.user.button.create
  - system.user.message.deleteSuccess
```

**Key 命名规则**:

| 规则 | 示例 | 说明 |
|------|------|------|
| 全部小写 | `common` | 不使用大写 |
| 多单词用点分隔 | `system.user` | 表达层级关系 |
| 语义清晰 | `search.placeholder` | 而非 `sp` |
| 避免缩写 | `button.create` | 而非 `btn.new` |

## 四、在模板中使用

### 4.1 基础用法

```vue
<script setup lang="ts">
import { useI18n } from 'vue-i18n'

const { t, locale } = useI18n()
</script>

<template>
  <!-- ✅ 推荐：使用全局注入的 $t -->
  <h1>{{ $t('system.user.title') }}</h1>

  <!-- 按钮文案 -->
  <el-button>{{ $t('common.operation.save') }}</el-button>
  <el-button type="danger">{{ $t('common.operation.delete') }}</el-button>

  <!-- 表单项 -->
  <el-form-item :label="$t('system.user.form.labels.userName')">
    <el-input :placeholder="$t('system.user.form.placeholders.userName')" />
  </el-form-item>

  <!-- 使用 useI18n Composition API -->
  <el-button @click="handleClick">{{ t('common.operation.edit') }}</el-button>
</template>
```

### 4.2 插值（变量替换）

```vue
<script setup lang="ts">
import { useI18n } from 'vue-i18n'

const { t } = useI18n()

// 语言包定义:
// "message": "用户 {name} 创建成功"

const userName = '张三'
</script>

<template>
  <!-- 输出: 用户 张三 创建成功 -->
  <p>{{ $t('system.user.message.createSuccess', { name: userName }) }}</p>

  <!-- 多个变量 -->
  <!-- "welcome": "欢迎，{name}！今天是 {date}" -->
  <p>{{ $t('common.welcome', { name: 'Admin', date: '2024-01-15' }) }}</p>
</template>
```

### 4.3 复数形式

```vue
<script setup lang="ts">
import { useI18n } from 'vue-i18n'

const { tc } = useI18n()

// 语言包定义（使用 | 分隔）
// "items": "0 项 | 1 项 | {n} 项"
</script>

<template>
  <!-- 数字自动匹配复数规则 -->
  <p>{{ tc('common.items', 0) }}</p>
  <!-- 输出: 0 项 -->
  <p>{{ tc('common.items', 1) }}</p>
  <!-- 输出: 1 项 -->
  <p>{{ tc('common.items', 5) }}</p>
  <!-- 输出: 5 项 -->
</template>
```

## 五、语言切换

### 5.1 语言切换组件

```vue
<!-- apps/admin/src/components/LanguageSwitch.vue -->
<script setup lang="ts">
import { computed } from 'vue'
import { useI18n } from 'vue-i18n'
import { ElDropdown, ElDropdownMenu, ElDropdownItem, ElButton } from 'element-plus'
import { SUPPORT_LOCALES, setLocale } from '@/locales'

const { locale } = useI18n()

const currentLanguage = computed(() => {
  return SUPPORT_LOCALES.find((l) => l.code === locale.value)
})

const handleCommand = (code: string) => {
  setLocale(code)
}
</script>

<template>
  <ElDropdown @command="handleCommand">
    <ElButton>
      {{ currentLanguage?.flag }} {{ currentLanguage?.name }}
      <el-icon class="el-icon--right"><ArrowDown /></el-icon>
    </ElButton>

    <template #dropdown>
      <ElDropdownMenu>
        <ElDropdownItem
          v-for="lang in SUPPORT_LOCALES"
          :key="lang.code"
          :command="lang.code"
          :disabled="lang.code === locale"
        >
          {{ lang.flag }} {{ lang.name }}
        </ElDropdownItem>
      </ElDropdownMenu>
    </template>
  </ElDropdown>
</template>
```

```typescript
    case 'zh-CN':

    default:
```

## 七、最佳实践

### 7.1 避免硬编码字符串

```vue
<!-- ❌ 避免：硬编码 -->
<el-button>新增用户</el-button>
<el-input placeholder="请输入用户名" />

<!-- ✅ 推荐：使用 $t -->
<el-button>{{ $t('system.user.button.create') }}</el-button>
<el-input :placeholder="$t('system.user.search.placeholder')" />
```

### 7.2 Key 保持一致

所有语言包的 key 结构必须完全一致：

```typescript
// ✅ 推荐：中英文 key 结构一致
// zh-CN
export default {
  operation: { confirm: '确认' }
}

// en
export default {
  operation: { confirm: 'Confirm' }
}
```

### 7.3 使用命名空间组织

```typescript
// ❌ 避免：扁平命名
"user_create_title": "新增用户",
"user_search_placeholder": "请输入用户名"

// ✅ 推荐：层次化
"system": {
  "user": {
    "form": {
      "title": { "create": "新增用户" }
    },
    "search": {
      "placeholder": "请输入用户名"
    }
  }
}
```

### 7.4 动态内容使用插值

```typescript
// ❌ 避免：在代码中拼接字符串
const message = '删除用户 ' + userName + ' 成功'

// ✅ 推荐：使用插值
const message = t('system.user.message.deleteSuccess', { userName })
// deleteSuccess: '删除用户 "{userName}" 成功'
```

### 7.5 避免嵌套过深

```typescript
// ❌ 避免：过深的嵌套
// "a.b.c.d.e.f.g": "..."

// ✅ 推荐：3-4 层较合适
// "system.user.title": "用户管理"
// "system.user.button.create": "新增用户"
```

### 7.6 处理特殊字符

```vue
<!-- 在模板中使用 HTML 字符 -->
<!-- "welcome": "欢迎 <strong>{name}</strong>" -->

<script setup lang="ts">
import { useI18n } from 'vue-i18n'

const { tm } = useI18n() // tm 用于获取原始消息模板（含 HTML）
</script>

<template>
  <!-- 使用 v-html 渲染（确保内容安全，无用户输入） -->
  <p v-html="$t('common.welcome', { name: 'Admin' })"></p>
</template>
```

## 九、测试与检查

### 9.1 Key 一致性检查（开发工具推荐）

在开发环境中，可通过 `missing` 回调打印缺失的 key：

```typescript
// apps/admin/src/locales/index.ts

export const i18n = createI18n({
  // ...
  missing: (locale, key) => {
    if (import.meta.env.DEV) {
      // 仅在开发环境警告
      console.warn(`[i18n] 缺失翻译: [${locale}] ${key}`)
    }
    return key
  }
})
```

### 9.2 单元测试

```typescript
// apps/admin/src/__tests__/i18n.test.ts

import { describe, it, expect } from 'vitest'
import { i18n, SUPPORT_LOCALES, setLocale } from '@/locales'

describe('i18n', () => {
  it('应该支持中英文切换', () => {
    setLocale('zh-CN')
    expect(i18n.global.t('common.operation.confirm')).toBe('确认')

    setLocale('en')
    expect(i18n.global.t('common.operation.confirm')).toBe('Confirm')
  })

  it('应该有一致的 key 结构', () => {
    const zh = i18n.global.messages['zh-CN'] as any
    const en = i18n.global.messages['en'] as any

    // 检查 common 下的 key 是否一致
    expect(Object.keys(zh.common)).toEqual(Object.keys(en.common))
  })
})
```
